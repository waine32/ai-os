You are a bash testing expert. Write a complete test suite for the following bash CLI script using bats-core (Bash Automated Testing System).

--- SCRIPT: ai ---
#!/bin/bash

set -euo pipefail

if [[ $# -lt 1 ]]; then
  echo "Usage: ai [flash|pro] [--session] \"prompt\"" >&2
  exit 1
fi

MODEL="$1"; shift

MODE="stateless"
if [[ "${1:-}" = "--session" ]]; then
  MODE="session"; shift
fi

PROMPT="${*:-}"
if [[ -z "$PROMPT" ]]; then
  echo "Error: prompt is required" >&2
  exit 1
fi

case "$MODEL" in
  flash)
    MODEL_NAME="deepseek-chat"
    ;;
  pro)
    MODEL_NAME="deepseek-reasoner"
    ;;
  *)
    echo "Usage: ai [flash|pro] [--session] \"prompt\"" >&2
    exit 1
    ;;
esac

ENDPOINT="https://api.deepseek.com/chat/completions"
API_KEY="${DEEPSEEK_API_KEY:?DEEPSEEK_API_KEY is not set}"
TEMPERATURE="${AI_TEMPERATURE:-0}"

MEMORY=""
if [[ -f ~/.ai-os/context/global.md ]]; then
  MEMORY=$(< ~/.ai-os/context/global.md)
fi

SYSTEM_PROMPT="You are a deterministic CLI assistant. Memory context:
$MEMORY"

USER_CONTENT="$PROMPT"
if [[ "$MODE" = "session" ]] && [[ -f ~/.ai-os/sessions/history.txt ]]; then
  HISTORY=$(< ~/.ai-os/sessions/history.txt)
  USER_CONTENT="${HISTORY}
USER: ${PROMPT}"
fi

PAYLOAD=$(jq -n \
  --arg model    "$MODEL_NAME" \
  --arg system   "$SYSTEM_PROMPT" \
  --arg user     "$USER_CONTENT" \
  --argjson temp "$TEMPERATURE" \
  '{model: $model, messages: [{role:"system",content:$system},{role:"user",content:$user}], temperature: $temp}')

RESPONSE=$(curl -s "$ENDPOINT" \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer $API_KEY" \
  -d "$PAYLOAD")

echo "$RESPONSE" > /tmp/ai_last_output.txt

OUTPUT=$(jq -r '.choices[0].message.content // empty' /tmp/ai_last_output.txt)

if [[ -z "$OUTPUT" ]]; then
  echo "Error: $(jq -r '.error.message // "no output – see /tmp/ai_last_output.txt"' /tmp/ai_last_output.txt)" >&2
  exit 1
fi

echo "$OUTPUT"

if [[ "$MODE" = "session" ]]; then
  mkdir -p ~/.ai-os/sessions
  printf 'USER: %s\nASSISTANT: %s\n' "$PROMPT" "$OUTPUT" >> ~/.ai-os/sessions/history.txt
fi
--- END SCRIPT ---
